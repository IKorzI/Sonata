ВВОДИТЬ В КОНСОЛЬ

=== Скрыть всё, кроме выделенного элемента в Elements ===
const style = document.createElement('style');
style.id = 'isolate-style';
style.innerHTML = '* { visibility: hidden !important; } .isolate-target, .isolate-target * { visibility: visible !important; }';
document.head.appendChild(style);
$0.classList.add('isolate-target');

=== Всё вернуть ===
document.getElementById('isolate-style').remove(); $0.classList.remove('isolate-target');
